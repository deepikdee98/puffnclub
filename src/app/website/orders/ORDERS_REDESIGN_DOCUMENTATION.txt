9.1 EXISTING APIs (Already Available)
--------------------------------------

GET /api/website/orders
Purpose: Get customer's orders
Query Params:
  - page: number
  - limit: number
  - status: string (optional filter)
Response:
{
  orders: Order[],
  pagination: {
    currentPage: number,
    totalPages: number,
    totalOrders: number,
    hasNextPage: boolean,
    hasPrevPage: boolean
  }
}

GET /api/website/orders/:orderId
Purpose: Get single order details
Response:
{
  order: Order
}

PUT /api/website/orders/:orderId/cancel
Purpose: Cancel order
Response:
{
  message: string,
  order: Order
}

9.2 NEW APIs NEEDED
-------------------

POST /api/website/orders/:orderId/review
Purpose: Submit product review/rating
Request Body:
{
  productId: string,
  rating: number (1-5),
  reviewText: string,
  customerName?: string
}
Response:
{
  message: string,
  review: {
    _id: string,
    product: string,
    customer: string,
    rating: number,
    comment: string,
    createdAt: string
  }
}

POST /api/website/orders/:orderId/exchange
Purpose: Submit exchange request
Request Body:
{
  productId: string,
  reason?: string,
  newSize: string,
  newColor: string,
  notes?: string
}
Response:
{
  message: string,
  exchangeRequest: {
    _id: string,
    order: string,
    product: string,
    status: 'pending' | 'approved' | 'rejected',
    newSize: string,
    newColor: string,
    createdAt: string
  }
}

POST /api/website/orders/:orderId/return
Purpose: Submit return request
Request Body:
{
  productId: string,
  reason: string,
  photos?: string[],
  bankDetails?: {
    accountNumber: string,
    ifscCode: string,
    accountHolderName: string
  },
  notes?: string
}
Response:
{
  message: string,
  returnRequest: {
    _id: string,
    order: string,
    product: string,
    status: 'pending' | 'approved' | 'rejected',
    reason: string,
    createdAt: string
  }
}

GET /api/website/orders/:orderId/tracking
Purpose: Get detailed tracking information
Response:
{
  order: Order,
  tracking: {
    currentStage: number (1-5),
    stages: [
      {
        name: string,
        status: 'completed' | 'active' | 'pending',
        timestamp?: string,
        location?: string,
        message?: string
      }
    ],
    estimatedDelivery?: string,
    carrier?: string,
    trackingNumber?: string
  }
}

POST /api/website/orders/:orderId/share
Purpose: Generate shareable tracking link
Response:
{
  shareUrl: string,
  expiresAt: string
}
